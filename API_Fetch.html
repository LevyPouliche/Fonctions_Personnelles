// Exemple d'une requête GET fetch('https://exemple.com/data') .then(response => { // Vérifie si la réponse HTTP est réussie (statut 200-299) if (!response.ok) { throw new Error('La requête a échoué avec le statut ' + response.status); } // Parse la réponse JSON return response.json(); }) .then(data => { // Utilise les données récupérées console.log(data); }) .catch(error => { // Gère les erreurs éventuelles console.error('Erreur lors de la requête :', error); });
